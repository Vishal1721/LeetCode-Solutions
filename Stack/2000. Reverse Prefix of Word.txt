class Solution {
public:
string revFun(int right, string& word) {
    int left=0;
    while(left<right) {
        swap(word[left],word[right]);
        left++;right--;
    }
    return word;
}
    string reversePrefix(string word, char ch) {
    for(int i=0;i<word.size();i++) {
        if(word[i]==ch) {
            return revFun(i,word);
        }
    }
    return word;
    }
};
