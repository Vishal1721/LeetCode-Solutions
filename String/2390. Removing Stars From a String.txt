class Solution {
public:
    string removeStars(string s) {
        stack<char>st;
        string k;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='*') {
                st.pop();
            }
            else {
                st.push(s[i]);
            }
        }
        while(!st.empty())
        {
            k.insert(k.begin(),st.top());
            st.pop();
        }
        
        return k;
    }
};
