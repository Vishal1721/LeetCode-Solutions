class Solution {
public:
    int lengthOfLastWord(string s) {
        int c=0;
        if(s.size()==1) return 1;
        for(int i=s.size()-1;i>=0;i--) {
            if((s[i]>='a'&&s[i]<='z') || (s[i]>='A'&&s[i]<='z')) {
                int j=i;
                while(s[j]!=' ') {
                    c++;
                    j--;
                }
                return c;
            }
        }
        return 0;
    }
};
