class Solution {
public:
    int numberOfSubstrings(string s) {
        vector<int>unSeen(3,-1);
        int cnt=0;
        for(int i=0;i<s.size();i++)
        {
            unSeen[s[i]-'a']=i;
            if(unSeen[0]!=-1 && unSeen[1]!=-1 && unSeen[2]!=-1 )
            {
                cnt=cnt+(1+min({unSeen[0],unSeen[1],unSeen[2]}));
            }
        }
        return cnt;
    }
};
//sliding window
