class Solution {
public:
    string addSpaces(string s, vector<int>& spaces) {
        int j=0;
        string k;
        for(int i=0;i<s.size();i++) {
            if(j<spaces.size() && i==spaces[j]) {
                k.push_back(' ');
                k.push_back(s[i]);
                j++;
            }
            else {
            k.push_back(s[i]);
            }
        }
        return k;
    }
};
