class Solution {
public:
    string frequencySort(string s) {
        unordered_map<char,int>mp;
        string k;
        for(int i=0;i<s.size();i++)
        {
            mp[s[i]]++;
        }
        vector<pair<char,int>>v(mp.begin(),mp.end());
        sort(v.begin(),v.end(),[](auto &a, auto &b)
        {
            return a.second>b.second;
        });

        for(auto &it:v)
        {
            k.append(it.second,it.first);
        }
    return k;
    }
};
