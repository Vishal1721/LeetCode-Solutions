class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        vector <int> arr(nums.size());
        int k=0,s=1;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]>=0)
            {
                arr[k]=nums[i];
                k=k+2;
            }
            else
            {
                arr[s]=nums[i];
                s=s+2;
            }
        }
        return arr;
    }
};
