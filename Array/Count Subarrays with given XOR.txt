class Solution {
  public:
    long subarrayXor(vector<int> &arr, int k) {
        // code here
        unordered_map<int,int>mp;
        mp[0]=1;
        int Xor=0,cnt=0;
        for(int i=0;i<arr.size();i++) {
            Xor^=arr[i];
            int ans = Xor^k;
            if(mp.find(ans)!=mp.end()) {
                cnt+=mp[ans];
            }
            mp[Xor]++;
        }
        return cnt;
    }
};
