class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int cost=0,maxProfit=0,mini=prices[0];
        for(int i=1;i<prices.size();i++)
        {
            cost=prices[i]-mini;
            maxProfit=max(maxProfit,cost);
            mini=min(mini,prices[i]);
        }
        return maxProfit;
    }
};
