class Solution {
public:
    vector<int> twoSum(vector<int>& n, int target) {
        int sum=0,j=n.size()-1,i=0;
        while(i<=j) {
            sum=n[i]+n[j];
            if(sum==target) {
                return {i+1,j+1};
            }
            else if(sum>target) {
                j--;
            }
            else
            {
                i++;
            }
        }
        return {-1,-1};
    }
};
//two pointer approach
