class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
    map<int,int>mpp;
    for(int i=0;i<nums.size();i++)
    {
        int sum=target-nums[i];
        if(mpp.find(sum)!=mpp.end())
        {
            return {i,mpp[sum]};
        }
        mpp[nums[i]]=i;

    }
    return {};
    }
};

//TC->O(Nlog n)
