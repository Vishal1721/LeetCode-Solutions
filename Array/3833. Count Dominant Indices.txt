class Solution {
public:
    int dominantIndices(vector<int>& nums) {
        int sum=0;
        for(int i=0;i<nums.size();i++) {
            sum=sum+nums[i];
        }
        int c=0;
        int n=nums.size();
        int prefSum=0;
        for(int i=0;i<nums.size()-1;i++) {
            prefSum+=nums[i];
            float ans=sum-prefSum;
            if((ans/(n-i-1))<nums[i]) c++;
        }
        return c;
    }
};
