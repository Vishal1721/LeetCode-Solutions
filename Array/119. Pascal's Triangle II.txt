class Solution {
public:
 vector<int> NcR(int n) {
        vector<int>ans;
        long long res=1;
        ans.push_back(res);
        for(int i=0;i<n;i++) {
            res=res*(n-i);
            res=res/(i+1);
            ans.push_back(res);
        }
    return ans;
 }
    vector<int> getRow(int rowIndex) {
        return NcR(rowIndex);
    }
};
