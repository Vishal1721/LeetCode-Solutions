class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& nums) {
        int i=0,j=nums.size()-1;
        while(i<j) {
            if(nums[i]%2==0) {
                nums[i]=nums[i];
                i++;
            }
            else {
                 swap(nums[i],nums[j]);
                 j--;
            }
            for(int i=0;i<nums.size();i++) {
                cout<<nums[i];
            }
            cout<<'\n';
        }
        return nums;
    }
};
