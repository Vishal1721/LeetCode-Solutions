class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& nums) {
      int low=0,high=nums.size()-1,mid=0;
      while(mid<high) {
        if(nums[mid]%2==1) {
            swap(nums[mid],nums[high]);
            high--;
        }
        else {
            low++;
            mid++;
        }
      }
      return nums;
    }
};
