class Solution {
  public:
    int maxSubarraySum(vector<int>& arr, int k) {
        int j=0,s=0,maxf=-1,sum=0;
        for(int i=0;i<arr.size();i++)
        {
            s++;
            sum=sum+arr[i];
            if(s==k)
            {
                maxf=max(maxf,sum);
                // cout<<sum<<endl;
                sum=sum-arr[j++];
            
                s--;
            }
        }
        return maxf;
    }
};
