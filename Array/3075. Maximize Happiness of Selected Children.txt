class Solution {
public:
    long long maximumHappinessSum(vector<int>& happiness, int k) {
        sort(happiness.begin(),happiness.end(),greater<>());
        long sum=0;
        int cnt=0;
        for(int i=0;i<k;i++) {
            if(happiness[i]-cnt<=0) {
                sum=sum+0;
                continue;
            }
            happiness[i]=happiness[i]-cnt;
            sum=sum+happiness[i];
            cnt++;
        }
        return sum;
    }
};
