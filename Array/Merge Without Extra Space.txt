class Solution {
  public:
    void mergeArrays(vector<int>& a, vector<int>& b) {
        // code here
        vector<int>ans;
        int i=a.size()-1,j=0;
        while(i>=0 && j<b.size()) {
            if(a[i]>b[j]) {
                swap(a[i],b[j]);
                i--;
                j++;
            }
            else {
                i--;j++;
            }
        }
       
        sort(a.begin(),a.end());
        sort(b.begin(),b.end());
       
    }
};
