class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int j=0,c=0;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]==0)
            {
                c++;
            }
            else if(nums[i]!=0)
            {
                nums[j]=nums[i];
                j++;
            }
        }
        while(c--)
        {
            nums[j]=0;
            j++;
        }
    }
};
